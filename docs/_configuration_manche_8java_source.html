332  @Override
333  public void handleMessage(@NonNull Message message)
334  {
335  // Log.d(TAG, "[Handler] id message = " + message.what);
336  // Log.d(TAG, "[Handler] message = " + message.obj.toString());
337 
338  switch(message.what)
339  {
340  case Communication.CONNEXION_BLUETOOTH:
341  Log.d(TAG, "[Handler] CONNEXION_BLUETOOTH");
342  actualiserEtatConnexionTable(true);
343  afficherConnexion(true, " Connexion : Succès ");
344  break;
345  case Communication.RECEPTION_BLUETOOTH:
346  Log.d(TAG, "[Handler] RECEPTION_BLUETOOTH");
347  Log.d(TAG, "message = 0x" + Integer.toHexString((int)message.obj));
348  break;
349  case Communication.DECONNEXION_BLUETOOTH:
350  Log.d(TAG, "[Handler] DECONNEXION_BLUETOOTH");
351  actualiserEtatConnexionTable(false);
352  break;
353  case Communication.ERREUR_BLUETOOTH:
354  Log.d(TAG, "[Handler] ERREUR_BLUETOOTH");
355  actualiserEtatConnexionTable(false);
356  afficherConnexion(false, " Connexion : Échec ");
357  break;
358  }
359  }
360  };
361  }
