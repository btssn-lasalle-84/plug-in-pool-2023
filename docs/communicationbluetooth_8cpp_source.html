191 bool CommunicationBluetooth::decoderTrame(const QStringList& champsTrame)
192 {
193  if(champsTrame.isEmpty())
194  return false;
195 
196  if(champsTrame.count() < NB_CHAMPS_MIN)
197  return false;
198 
199  qDebug() << Q_FUNC_INFO << "champsTrame" << champsTrame;
200 
201  QString type = champsTrame.at(POSITION_TYPE);
202 
203  if(type.isEmpty())
204  return false;
205 
206  switch(type.at(POSITION_TYPE + 1).toLatin1())
207  {
208  case TYPE_EMPOCHAGE:
209  {
210  int table = champsTrame.value(POSITION_TABLE).toInt();
211  int poche = champsTrame.value(POSITION_POCHE).toInt();
212  int couleur = champsTrame.value(POSITION_COULEUR).toInt();
213  qDebug() << Q_FUNC_INFO << "type"
214  << type.at(POSITION_TYPE + 1).toLatin1() << "table"
215  << table << "poche" << poche << "couleur" << couleur;
216  emit empochage(table, poche, couleur);
217  break;
218  }
219  case TYPE_NOMS:
220  {
221  int table = champsTrame.value(POSITION_TABLE).toInt();
222  QString joueur1 = champsTrame.value(POSITION_JOUEUR1);
223  QString joueur2 = champsTrame.value(POSITION_JOUEUR2);
224  qDebug() << Q_FUNC_INFO << "type"
225  << type.at(POSITION_TYPE + 1).toLatin1() << "table"
226  << table << "joueur1" << joueur1 << "joueur2" << joueur2;
227  emit nomsJoueurs(table, joueur1, joueur2);
228  break;
229  }
230  case TYPE_CHANGEMENT:
231  {
232  int table = champsTrame.value(POSITION_TABLE).toInt();
233  int changement = champsTrame.value(POSITION_CHANGEMENT).toInt();
234  qDebug() << Q_FUNC_INFO << "type"
235  << type.at(POSITION_TYPE + 1).toLatin1() << "table"
236  << table << "changement" << changement;
237  emit changementJoueur(table, changement);
238  break;
239  }
240  default:
241  qDebug() << Q_FUNC_INFO << "type inconnu";
242  return false;
243  break;
244  }
245  return true;
246 }
