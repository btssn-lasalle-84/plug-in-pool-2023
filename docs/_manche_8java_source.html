376  private void initialiserHandler()
377  {
378  this.handler = new Handler(this.getMainLooper())
379  {
380  @Override
381  public void handleMessage(@NonNull Message message)
382  {
383  // Log.d(TAG, "[Handler] id message = " + message.what);
384  // Log.d(TAG, "[Handler] message = " + message.obj.toString());
385 
386  switch(message.what)
387  {
388  case Communication.CONNEXION_BLUETOOTH:
389  Log.d(TAG, "[Handler] CONNEXION_BLUETOOTH");
390  connexionTable = true;
391  break;
392  case Communication.RECEPTION_BLUETOOTH:
393  Log.d(TAG, "[Handler] RECEPTION_BLUETOOTH");
394  Log.d(TAG, "message = 0x" + Integer.toHexString((int)message.obj));
395  byte trame = ((Integer)message.obj).byteValue();
396  //Log.d(TAG, "trame = " + ProtocoleTable.byteToBinaryString(trame));
397  Log.d(BlackBall.TAG, "trame = " + ProtocoleTable.byteToBinaryString(trame));
398  if((trame & ProtocoleTable.MASQUE_TYPE) != 0)
399  {
400  traiterTrameService(trame);
401  }
402  else
403  {
404  int couleur = (int)(trame & ProtocoleTable.MASQUE_COULEUR);
405  int poche = (int)((trame & ProtocoleTable.MASQUE_POCHE) >> ProtocoleTable.CHAMP_POCHE);
406  int[] empoche = {poche, couleur};
407  manche.get(manche.size() - 1).add(empoche);
408 
409  if(communications[Communication.ECRAN] != null)
410  {
411  String trameEnvoie = "" + ProtocoleEcran.DELIMITEUR_DEBUT + ProtocoleEcran.TYPE_EMPOCHE + ProtocoleEcran.DELIMITEUR_CHAMPS + ProtocoleEcran.TABLES.charAt(Character.getNumericValue(table.charAt(CHAR_NUMERO_TABLE)) - 1) + ProtocoleEcran.DELIMITEUR_CHAMPS + ProtocoleEcran.POCHES.charAt(poche) + ProtocoleEcran.DELIMITEUR_CHAMPS + ProtocoleEcran.COULEURS.charAt(couleur) + ProtocoleEcran.DELIMITEUR_FIN;
412  communications[Communication.ECRAN].envoyer(trameEnvoie);
413  Log.d(ProtocoleEcran.TAG,trameEnvoie);
414  }
415  if(couleur == BlackBall.NOIRE)
416  {
417  empocherBilleNoire();
418  }
419  else if(couleur == BlackBall.BLANCHE)
420  {
421  empocherBilleBlanche();
422  }
423  else
424  {
425  empocherBilleCouleur(poche, couleur);
426  }
427  }
428  break;
429  case Communication.DECONNEXION_BLUETOOTH:
430  Log.d(TAG, "[Handler] DECONNEXION_BLUETOOTH");
431  connexionTable = false;
432  break;
433  }
434  }
435  };
436  }
